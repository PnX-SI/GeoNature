Description

Ce composant permet de slectionner une ou plusieurs zones gographiques.

Il encapsule ng-select et utilise un observable contenant une liste de
zones gographiques qui peuvent tre restreint par une liste de types (typeCodes).

La liste des zones gographiques correspond aux caractres taps dans la
zone de recherche. L'affichage de la liste ne se dclenche qu' partir de la
saisie du second caractres.
Par dfaut, la liste des 100 premires zones gographiques et pr-charges.

Ce composant retourne une liste d'identifiant de zones gographiques (id_areas)
dans la variable parentFormControl correspondant aux slections de l'utilisateur.

En mode mise jour, il est ncessaire de fournir dans parentFormControl une liste
d'id de zones go. Ces id seront compars l'aide de la fonction de l'input
compareWith aux objets passs dans l'input defaultItems.

Example

<pnx-areas
label="Mon libell de champ"
[typeCodes]="['COM', 'DEP']"
[parentFormControl]="form.controls.areas"
>
</pnx-areas>

import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';

import { Subject, Observable, of, concat, zip } from 'rxjs';
import {
  distinctUntilChanged,
  debounceTime,
  switchMap,
  tap,
  catchError,
  map,
} from 'rxjs/operators';

import { DataFormService } from '../data-form.service';
import { GenericFormComponent } from '@geonature_common/form/genericForm.component';

/**
 * Ce composant permet de slectionner une ou plusieurs zones gographiques.
 *
 * Il encapsule ng-select et utilise un observable contenant une liste de
 * zones gographiques qui peuvent tre restreint par une liste de types (`typeCodes`).
 *
 * La liste des zones gographiques correspond aux caractres taps dans la
 * zone de recherche. L'affichage de la liste ne se dclenche qu' partir de la
 * saisie du second caractres.
 * Par dfaut, la liste des 100 premires zones gographiques et pr-charges.
 *
 * Ce composant retourne une liste d'identifiant de zones gographiques (`id_areas`)
 * dans la variable `parentFormControl` correspondant aux slections de l'utilisateur.
 *
 * En mode mise  jour, il est ncessaire de fournir dans `parentFormControl` une liste
 * d'id de zones go. Ces id seront compars  l'aide de la fonction de l'input
 * `compareWith` aux objets passs dans l'input `defaultItems`.
 *
 * @example
 * <pnx-areas
 *   label="Mon libell de champ"
 *   [typeCodes]="['COM', 'DEP']"
 *   [parentFormControl]="form.controls.areas"
 * >
 * </pnx-areas>
 */
@Component({
  selector: 'pnx-areas',
  templateUrl: 'areas.component.html',
})
export class AreasComponent extends GenericFormComponent implements OnInit {
  /**
   * Permet de dfinir une liste de `type_code`  prendre compte.
   * Utiliser une valeur du champ "`type_code`" de la table "`bib_areas_types`".
   * Les zones gographiques affiches dans la liste sont limites aux
   * `type_code` indiqus dans cet attribut.
   */
  @Input() typeCodes: Array<string> = []; // Areas type_code
  /**
   * Do not get the geom from the getArea call, for performances
   */
  @Input() withoutGeom: Boolean = false;
  /**
   * Nom du champ  utiliser pour dterminer la valeur  utiliser pour le
   * contenu du `FormControl`.
   * Si vous souhaitez forcer le maintient dans `parentFormControl` d'un
   * tableau d'objets comprenant le champ identifiant et le champ pour
   * l'affichage (`area_name`), vous pouvez l'indiquer en passant une
   * valeur `null`  cet attribut.
   * Exemple :
   * [valueFieldName]="'id_type'"
   */
  @Input() valueFieldName: string = 'id_area';
  /**
   * Fonction de comparaison entre les lements slectionns prsent dans
   * le `parentFormControl` et les lments affichs dans la liste des options.
   * @param item
   * @param selected
   * @returns
   */
  @Input() compareWith = (item, selected) => item[this.valueFieldName] === selected;
  /**
   * Tableau d'objets qui doivent contenir chacun  minima 2 attributs :
   *  - un attribut correspondant  la valeur de l'input `valueFieldName` et
   * contenant l'identifiant de la zone gographique.
   *  - un attribut `area_name` contenant l'intitul de la zone gographique
   *  afficher.
   * En mise  jour, ces objets doivent correspondres aux id prsents dans
   * `parentFormControl`.
   */
  @Input() defaultItems: Array<any> = [];
  /**
   * Permet dcouter les changements sur la slection de ng-select.
   * Retourne un tableau d'objets. Les objets correspondent aux items
   * slectionns. Chaque objet contient  minima 2 attributs : un
   * correspondant  l'input `valueFieldName`, l'autre est `displayName`.
   */
  @Output() onSelectionChange = new EventEmitter<any>();
  areas_input$ = new Subject<string>();
  areas: Observable<any>;
  loading = false;
  params = {};

  constructor(private dataService: DataFormService) {
    super();
  }

  ngOnInit() {
    super.ngOnInit();
    this.params['type_code'] = this.typeCodes;
    this.params['without_geom'] = this.withoutGeom;
    this.valueFieldName = this.valueFieldName === undefined ? 'id_area' : this.valueFieldName;

    this.getAreas();
  }

  /**
   * Merge initial 100 areas + default values (for update)
   */
  initialAreas(): Observable<any> {
    return zip(
      this.dataService.getAreas(this.params).pipe(map((data) => this.formatAreas(data))), // Default items
      of(this.defaultItems) // Default items in update mode
    ).pipe(
      map((areasArrays) => {
        // Remove dubplicates items
        const items = areasArrays[0];
        const defaultItems = areasArrays[1];
        if (defaultItems && defaultItems.length > 0) {
          const filteredItems = items.filter((area) => {
            return !defaultItems.some(
              (defaultArea) => defaultArea[this.valueFieldName] === area[this.valueFieldName]
            );
          });
          return filteredItems.concat(defaultItems);
        } else {
          return items;
        }
      })
    );
  }

  getAreas() {
    this.areas = concat(
      this.initialAreas(),
      this.areas_input$.pipe(
        debounceTime(200),
        distinctUntilChanged(),
        tap(() => (this.loading = true)),
        switchMap((term) => {
          this.params['area_name'] = term;
          return term && term.length >= 2
            ? this.dataService.getAreas(this.params).pipe(
                map((data) => this.formatAreas(data)),
                catchError(() => of([])), // Empty list on error
                tap(() => (this.loading = false))
              )
            : of([]);
        }),
        tap(() => (this.loading = false))
      )
    );
  }

  /**
   * Ajouter entre parenthse le numro du dpartement si le premier objet
   * contient un champ `id_type` correspondant au type commune (=COM).
   * @param data Liste d'objets contenant des infos sur des zones gographiques.
   */
  private formatAreas(data: Partial<{ id_type: number; area_code: string }>[]) {
    if (data.length > 0 && data[0]['area_type']['type_code'] === 'COM') {
      return data.map((element) => {
        element['area_name'] = `${element['area_name']} (${element.area_code.substring(0, 2)}) `;
        return element;
      });
    }
    return data;
  }
}
