Il encapsule le composant pnx-multiselect
et lui passe une liste de noms scientifiques suprieurs un rang donn.

La liste de noms scientifiques correspond aux caractres taps dans la

rank="GN"
[parentFormControl]="form.controls.taxa"
[bindAllItem]="true"

import { Component, OnInit, Input } from '@angular/core';
import { DataFormService } from '../data-form.service';
import { UntypedFormControl } from '@angular/forms';
import { CommonService } from '@geonature_common/service/common.service';

/**
 * Ce composant a le mme comportement que pnx-areas.
 *
 * Il encapsule [le composant pnx-multiselect]{@link MultiSelectComponent}
 * et lui passe une liste de noms scientifiques suprieurs  un rang donn.
 *
 * La liste de noms scientifiques correspond aux caractres taps dans la
 * zone de recherche.
 *
 * Au maximum 20 noms sont affichs.
 *
 * @example
 * <pnx-taxa
 *   label="Mon libell de champ"
 *   rank="GN"
 *   [parentFormControl]="form.controls.taxa"
 *   [bindAllItem]="true"
 *   [debounceTime]="400"
 * >
 * </pnx-taxa>
 */
@Component({
  selector: 'pnx-taxa',
  templateUrl: 'taxa.component.html',
})
export class TaxaComponent implements OnInit {
  /** Contient un tableau d'objets. L'attribut "displayName" des objets
   * sert  l'affichage du nom.
   */
  public taxa: any;
  /** Rang minimum des noms recherchs.
   * Utiliser une valeur du champ "id_rang" de la table "bib_taxref_rangs".
   *
   * Par dfaut, le rang dfini correspond au Genre (=GN).
   */
  @Input() rank: string = 'GN';
  /** Intitul du champ. */
  @Input() label: string;
  /** Passer une instance de FormControl afin de pouvoir valider le
   * formulaire et rcuprer la valeur de ce champ.
   */
  @Input() parentFormControl: UntypedFormControl;
  /** Indique si tout l'objet correspondant  la slection doit tre
   * pass au formControl (=true) ou seulement la valeur de la proprit
   * "cd_nom" (=false).
   *
   * Par dfaut, seul la valeur de "cd_nom" est transmise (=false).
   */
  @Input() bindAllItem: false;
  /** Temps en millisecondes avant lancement d'une recherche avec les
   * caractres saisis dans la zone de recherche.
   *
   * Voir le composant [pnx-multiselect]{@link MultiSelectComponent} pour
   * la valeur par dfaut.
   */
  @Input() debounceTime: number;
  /** Dsactive le composant. */
  @Input() disabled: boolean = false;

  /** @ignore */
  constructor(
    private dataService: DataFormService,
    private commonService: CommonService
  ) {}

  ngOnInit() {
    this.taxa = [];

    // In update mode
    if (this.parentFormControl.value) {
      this.taxa = this.parentFormControl.value;
    }
  }

  /** Relance un appel  l'API fournissant les noms scientifiques si le
   * nombre de caractres saisies dans le zone de recherche est suprieur
   * ou gal  2.
   *
   * La liste de noms est rinitialise quand un nom est slectionn ou
   * la zone de recherche vide.
   *
   * @param taxon_name La chaine de caractre saisie dans la zone de
   * recherche.
   */
  refreshTaxaList(taxon_name) {
    if (taxon_name && taxon_name.length >= 2) {
      this.dataService.getHigherTaxa(this.rank, taxon_name).subscribe(
        (data) => {
          this.taxa = data;
        },
        (err) => {
          if (err.status === 404) {
            this.taxa = [{ displayName: 'No data to display' }];
          } else {
            this.taxa = [];
          }
        }
      );
    } else if (!taxon_name) {
      this.taxa = [];
    }
  }
}
